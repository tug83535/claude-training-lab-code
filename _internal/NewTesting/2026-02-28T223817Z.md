## SECTION 2 — ALL DISCOVERED ISSUES

### 2.1 Floating-Point Precision (Critical)

**What was found:**
- `P&L - Monthly Trend`: **318 cells** with floating-point anomalies
  - Examples: `9412.300000000001`, `9693.210000000001`, `9642.119999999999`
  - FY Total column: `28329.980000000003`, `29500.370000000003`
- `Charts & Visuals` helper table rows 201–220: **139 cells** with anomalies
  - Examples: `6180.139999999999`, `6712.750000000001`, `1448.8999999999996`
  - Margin ratio row 204: `0.5419585695099759`, `0.5594830807367375`
- `Report-->` Executive Summary:
  - D8 Contribution Margin: `0.5787034647912902` (should be `57.9%`)
  - F11 Q4 vs Q1 Growth: `0.1985582058300074` (should be `19.9%`)
  - B8 Full Year Revenue: `122468.56` displayed correctly but raw value unrounded

**Root cause:** AI generated formulas using raw arithmetic without `ROUND()` wrappers.
IEEE 754 binary floating-point accumulates tiny errors through SUM chains.

**Fix required:** Wrap all currency calculations in `=ROUND(expr, 2)`.
Wrap all percentage calculations in `=ROUND(expr, 4)`.
Add a named constant `FP_TOLERANCE = 0.005` for use in all reconciliation checks.

### Macro 1: `RunReconciliation`
**Purpose:** Recalculates the Checks tab, timestamps it, and color-codes every PASS/FAIL row.
Shows a popup with the count of failures.

```vb
Public Sub RunReconciliation()
    Dim wsChk As Worksheet
    Set wsChk = ThisWorkbook.Sheets("Checks")

    wsChk.Calculate
    wsChk.Range("E1").Value = "Last Run: " & Format(Now, "yyyy-mm-dd hh:nn:ss")

    Dim lastRow As Long
    lastRow = wsChk.Cells(wsChk.Rows.Count, "A").End(xlUp).Row

    Dim failCount As Long: failCount = 0
    Dim r As Long
    For r = 5 To lastRow
        If UCase(wsChk.Cells(r, 5).Value) = "FAIL" Then
            wsChk.Cells(r, 5).Interior.Color = RGB(255, 200, 200)
            failCount = failCount + 1
        Else
            wsChk.Cells(r, 5).Interior.Color = RGB(200, 255, 200)
        End If
    Next r

    If failCount > 0 Then
        MsgBox failCount & " check(s) FAILED. Review the Checks tab.", _
               vbExclamation, "Reconciliation"
    Else
        MsgBox "All checks PASSED.", vbInformation, "Reconciliation"
    End If
End Sub
```

### Macro 2: `FixTextNumbers`
**Purpose:** Scans Assumptions column B and converts any cell where the value is a string
that passes IsNumeric() into a proper Double. Applies `0.0%` format. Fixes the `"0.22"` bug.

```vb
Public Sub FixTextNumbers()
    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets("Assumptions")

    Dim lastRow As Long
    lastRow = ws.Cells(ws.Rows.Count, "B").End(xlUp).Row

    Dim fixCount As Long: fixCount = 0
    Dim r As Long
    For r = 1 To lastRow
        Dim c As Range: Set c = ws.Cells(r, 2)
        If Not IsEmpty(c.Value) Then
            If IsNumeric(c.Value) And VarType(c.Value) = vbString Then
                c.Value = CDbl(c.Value)
                c.NumberFormat = "0.0%"
                fixCount = fixCount + 1
            End If
        End If
    Next r

    MsgBox fixCount & " text-number(s) converted to numeric.", _
           vbInformation, "Fix Text Numbers"
End Sub
```

### Macro 3: `CleanFloatingPoint`
**Purpose:** Scans all numeric constant cells on the active sheet. If the decimal portion
extends beyond 4 digits (indicating floating-point noise), rounds to 2 decimal places.

```vb
Public Sub CleanFloatingPoint()
    Dim ws As Worksheet: Set ws = ActiveSheet
    Dim rng As Range

    On Error Resume Next
    Set rng = ws.UsedRange.SpecialCells(xlCellTypeConstants, xlNumbers)
    On Error GoTo 0

    If rng Is Nothing Then
        MsgBox "No numeric constants found.", vbInformation, "FP Cleaner"
        Exit Sub
    End If

    Dim cell As Range
    Dim cleanCount As Long: cleanCount = 0
    For Each cell In rng
        If Len(CStr(cell.Value)) - InStr(CStr(cell.Value), ".") > 4 Then
            cell.Value = Round(cell.Value, 2)
            cleanCount = cleanCount + 1
        End If
    Next cell

    MsgBox cleanCount & " cell(s) rounded to 2dp on '" & _
           ws.Name & "'.", vbInformation, "FP Cleaner"
End Sub
```

### Script Name: `kbt_etl_pipeline.py`
**Requirements:** `pip install openpyxl pandas`
**Usage:** `python kbt_etl_pipeline.py <input.xlsx> [--output KBT_Cleaned.xlsx]`
**Output files:** `KBT_Cleaned.xlsx` (3 sheets) + `KBT_Cleaned.dq_log.txt`

**What it produces:**

- Sheet `DQ_Reconciliation`: 17 automated checks with PASS/FAIL/INFO status
- Sheet `SummaryPivot`: Aggregated by Year, Quarter, Month, Department, Product

```python
#!/usr/bin/env python3
"""
Keystone BenefitTech — P&L Model ETL & Data Quality Pipeline
=============================================================
Reads CrossfireHiddenWorksheet, cleans it, validates it,
and writes a Power-Query-ready staging file.

Requirements: pip install openpyxl pandas
Run:          python kbt_etl_pipeline.py <input.xlsx> [--output cleaned.xlsx]
"""

import argparse
import sys
from pathlib import Path
from datetime import datetime
from collections import Counter

import pandas as pd
import openpyxl


# ── CONFIGURATION ────────────────────────────────────────────────────
HIDDEN_SHEET     = "CrossfireHiddenWorksheet"
EXPECTED_COLS    = ["ID", "Date", "Department", "Product",
                    "Expense Category", "Vendor", "Amount"]
VALID_DEPTS      = {"NetOps", "Security", "Support", "Partners",
                    "Content", "R&D", "Product Management"}
VALID_PRODUCTS   = {"iGO", "Affirm", "InsureSight", "DocFast"}
DATE_FORMATS     = ["%Y-%m-%d", "%m/%d/%Y", "%Y-%m-%d %H:%M:%S",
                    "%-m/%-d/%Y", "%m-%d-%Y"]
AMOUNT_TOLERANCE = 0.005

LOG        = []
WARN_COUNT = 0
FAIL_COUNT = 0


def log(severity: str, msg: str) -> None:
    global WARN_COUNT, FAIL_COUNT
    ts = datetime.now().strftime("%H:%M:%S")
    entry = f"[{ts}] {severity:5s} | {msg}"
    LOG.append(entry)
    if severity == "WARN":
        WARN_COUNT += 1
    elif severity == "FAIL":
        FAIL_COUNT += 1
    print(entry)


# ── EXTRACT ──────────────────────────────────────────────────────────
def extract(path: str) -> pd.DataFrame:
    wb = openpyxl.load_workbook(path, data_only=True)
    if HIDDEN_SHEET not in wb.sheetnames:
        log("FAIL", f"Sheet '{HIDDEN_SHEET}' not found.")
        sys.exit(1)

    ws   = wb[HIDDEN_SHEET]
    data = []
    for row in ws.iter_rows(min_row=2, max_row=ws.max_row, values_only=True):
        data.append(list(row))

    df = pd.DataFrame(data, columns=EXPECTED_COLS)
    log("INFO", f"Extracted {len(df)} raw rows from '{HIDDEN_SHEET}'.")
    return df


# ── TRANSFORM / CLEAN ────────────────────────────────────────────────
def parse_date(val):
    """Coerce any date representation to a proper datetime."""
    if isinstance(val, datetime):
        return val
    if isinstance(val, str):
        for fmt in DATE_FORMATS:
            try:
                return datetime.strptime(val.strip(), fmt)
            except ValueError:
                continue
    return pd.NaT


def clean(df: pd.DataFrame) -> pd.DataFrame:
    original_len = len(df)

    # Dates
    text_dates = df["Date"].apply(lambda x: isinstance(x, str)).sum()
    if text_dates:
        log("WARN", f"{text_dates} date(s) stored as text — coercing.")
    df["Date"]    = df["Date"].apply(parse_date)
    bad_dates     = df["Date"].isna().sum()
    if bad_dates:
        log("FAIL", f"{bad_dates} unparseable date(s) dropped.")
        df = df.dropna(subset=["Date"])

    df["Month"]   = df["Date"].dt.month
    df["Quarter"] = df["Date"].dt.quarter
    df["Year"]    = df["Date"].dt.year

    # Amounts
    text_amts = df["Amount"].apply(lambda x: isinstance(x, str)).sum()
    if text_amts:
        log("WARN", f"{text_amts} amount(s) stored as text — coercing.")
    df["Amount"] = pd.to_numeric(df["Amount"], errors="coerce")
    bad_amts     = df["Amount"].isna().sum()
    if bad_amts:
        log("FAIL", f"{bad_amts} non-numeric amount(s) dropped.")
        df = df.dropna(subset=["Amount"])
    df["Amount"] = df["Amount"].round(2)

    # Duplicates
    dup_mask = df.duplicated(subset="ID", keep="first")
    n_dupes  = dup_mask.sum()
    if n_dupes:
        log("WARN", f"{n_dupes} duplicate ID(s) removed (kept first).")
        df = df[~dup_mask].copy()

    # Domain validation
    bad_dept = ~df["Department"].isin(VALID_DEPTS)
    if bad_dept.any():
        log("FAIL", f"{bad_dept.sum()} row(s) with invalid department.")
    bad_prod = ~df["Product"].isin(VALID_PRODUCTS)
    if bad_prod.any():
        log("FAIL", f"{bad_prod.sum()} row(s) with invalid product.")

    log("INFO", f"Clean: {original_len} → {len(df)} rows "
                f"(removed {original_len - len(df)}).")
    return df.reset_index(drop=True)


# ── VALIDATE ─────────────────────────────────────────────────────────
def validate(df: pd.DataFrame) -> pd.DataFrame:
    checks = []

    # Revenue share
    rev_shares = 0.55 + 0.28 + 0.12 + 0.05
    diff       = abs(rev_shares - 1.0)
    checks.append({
        "Check"      : "Revenue Share %s sum to 100%",
        "Value_A"    : rev_shares,
        "Value_B"    : 1.0,
        "Difference" : diff,
        "Status"     : "PASS" if diff < AMOUNT_TOLERANCE else "FAIL"
    })

    # Per-product expense totals (informational)
    for prod in VALID_PRODUCTS:
        total = df.loc[df["Product"] == prod, "Amount"].sum()
        checks.append({
            "Check"      : f"Total expenses — {prod}",
            "Value_A"    : round(total, 2),
            "Value_B"    : "informational",
            "Difference" : 0,
            "Status"     : "INFO"
        })

    # Monthly totals > 0
    monthly = df.groupby("Month")["Amount"].sum()
    for m, amt in monthly.items():
        checks.append({
            "Check"      : f"Month {m:02d} total > 0",
            "Value_A"    : round(amt, 2),
            "Value_B"    : "> 0",
            "Difference" : 0 if amt > 0 else amt,
            "Status"     : "PASS" if amt > 0 else "FAIL"
        })

    # Row count
    expected = 500
    actual   = len(df)
    checks.append({
        "Check"      : f"Row count = {expected} after dedup",
        "Value_A"    : actual,
        "Value_B"    : expected,
        "Difference" : abs(actual - expected),
        "Status"     : "PASS" if actual == expected else "WARN"
    })

    recon = pd.DataFrame(checks)
    log("INFO", f"Validation complete — {len(checks)} checks run.")
    return recon


# ── LOAD ─────────────────────────────────────────────────────────────
def load(df: pd.DataFrame, recon: pd.DataFrame, out_path: str) -> None:
    with pd.ExcelWriter(out_path, engine="openpyxl") as writer:
        df.to_excel(writer, sheet_name="CleanedTransactions", index=False)
        recon.to_excel(writer, sheet_name="DQ_Reconciliation", index=False)

        pivot = (df.groupby(
                    ["Year", "Quarter", "Month", "Department", "Product"])
                   ["Amount"].sum()
                   .reset_index())
        pivot["Amount"] = pivot["Amount"].round(2)
        pivot.to_excel(writer, sheet_name="SummaryPivot", index=False)

    log("INFO", f"Output written to '{out_path}'.")
    log("INFO", f"DQ Summary: {WARN_COUNT} warnings, {FAIL_COUNT} failures.")


# ── MAIN ─────────────────────────────────────────────────────────────
def main():
    parser = argparse.ArgumentParser(
        description="Keystone BenefitTech P&L ETL Pipeline")
    parser.add_argument("input",   help="Path to the source .xlsx")
    parser.add_argument("--output", default="KBT_Cleaned.xlsx")
```
