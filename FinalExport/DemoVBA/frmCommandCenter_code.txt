'===============================================================================
' LoadCategories - Populate the category list (left panel)
'===============================================================================
Private Sub LoadCategories()
    lstCategories.Clear
    lstCategories.AddItem "All Actions"
    Dim cats As Variant
    cats = Array("Monthly Operations", "Analysis", "Data Quality", _
        "Reporting", "Utilities", "Data & Import", "Forecasting", _
        "Scenarios", "Allocation", "Consolidation", _
        "Version Control", "Governance", "Admin & Testing", "Advanced", _
        "Sheet Tools", "What-If Demo")
    Dim c As Variant
    For Each c In cats
        lstCategories.AddItem CStr(c)
    Next c
End Sub

'===============================================================================
' Event Handlers - Category selection and search trigger filtering
'===============================================================================
Private Sub lstCategories_Click()
    FilterActions
End Sub

Private Sub txtSearch_Change()
    FilterActions
End Sub

'===============================================================================
' FilterActions - Apply category + search text filters to action list
'===============================================================================
Private Sub FilterActions()
    lstActions.Clear
    Dim selCat As String
    If lstCategories.ListIndex >= 0 Then
        selCat = lstCategories.List(lstCategories.ListIndex)
    Else
        selCat = "All Actions"
    End If
    Dim srch As String: srch = LCase(Trim(txtSearch.Text))
    Dim cnt As Long: cnt = 0
    ReDim m_FilteredNums(1 To 65)
    Dim i As Long
    For i = 1 To m_ActionCount
        Dim catMatch As Boolean
        catMatch = (selCat = "All Actions") Or (m_Actions(i).Category = selCat)
        Dim srchMatch As Boolean
        srchMatch = (srch = "") Or _
            (InStr(1, LCase(m_Actions(i).Label), srch) > 0) Or _
            (InStr(1, LCase(m_Actions(i).Category), srch) > 0)
        If catMatch And srchMatch Then
            cnt = cnt + 1
            m_FilteredNums(cnt) = m_Actions(i).Num
            lstActions.AddItem m_Actions(i).Num
            lstActions.List(lstActions.ListCount - 1, 1) = m_Actions(i).Label
        End If
    Next i
    lblStatus.Caption = cnt & " actions shown"
    If cnt > 0 Then lstActions.ListIndex = 0
End Sub

'===============================================================================
' Double-click an action to run it immediately
'===============================================================================
Private Sub lstActions_DblClick(ByVal Cancel As MSForms.ReturnBoolean)
    RunSelected
End Sub

'===============================================================================
' Button Handlers
'===============================================================================
Private Sub btnRun_Click()
    RunSelected
End Sub

Private Sub btnRunClose_Click()
    Dim actNum As Long: actNum = GetSelectedNum()
    If actNum = 0 Then Exit Sub
    Me.Hide
    modFormBuilder.ExecuteAction actNum
    Unload Me
End Sub

Private Sub btnClose_Click()
    Unload Me
End Sub

'===============================================================================
' RunSelected - Execute the currently highlighted action, then return to form
'===============================================================================
Private Sub RunSelected()
    Dim actNum As Long: actNum = GetSelectedNum()
    If actNum = 0 Then Exit Sub
    lblStatus.Caption = "Running #" & actNum & "..."
    Me.Hide
    modFormBuilder.ExecuteAction actNum
    Me.Show
    lblStatus.Caption = "Last run: #" & actNum & " - Done"
End Sub

'===============================================================================
' GetSelectedNum - Return the action number of the selected list item
'===============================================================================
Private Function GetSelectedNum() As Long
    GetSelectedNum = 0
    If lstActions.ListIndex < 0 Then
        MsgBox "Select an action first.", vbExclamation
        Exit Function
    End If
    GetSelectedNum = CLng(lstActions.List(lstActions.ListIndex, 0))
End Function

'===============================================================================
' Escape key / X button closes the form
'===============================================================================
Private Sub UserForm_QueryClose(Cancel As Integer, CloseMode As Integer)
    If CloseMode = 0 Then Unload Me
End Sub
